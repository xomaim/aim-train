    <script>
        // State
        let currentUser = null;
        let currentAuthTab = 'login';
        let currentGame = null;
        let gameRunning = false;

            score: 0, hits: 0, misses: 0, time: 60,
            startTime: null, targets: [], particles: [],
            mouseX: 0, mouseY: 0, isMouseDown: false,
            isScoped: false
        };
        let userStats = {
            totalScore: 0, totalHits: 0, totalMisses: 0, gamesPlayed: 0
        };
        let sensConfig = {
            dpi: 800, hip: 1.20, ads: 0.80, scope: 0.50
        };

        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        let animationId;

        function resizeCanvas() {
            const container = document.getElementById('canvasContainer');
            if (container) {
                canvas.width = container.clientWidth;
                canvas.height = container.clientHeight;
            }
        }
        window.addEventListener('resize', resizeCanvas);

        // Sensitivity Math
        function updateSens() {
            sensConfig.dpi = parseInt(document.getElementById('dpiSlider').value);
            sensConfig.hip = parseFloat(document.getElementById('hipSlider').value);
            sensConfig.ads = parseFloat(document.getElementById('adsSlider').value);
            sensConfig.scope = parseFloat(document.getElementById('scopeSlider').value);
            
            document.getElementById('dpiVal').textContent = sensConfig.dpi;
            document.getElementById('hipVal').textContent = sensConfig.hip.toFixed(2);
            document.getElementById('adsVal').textContent = sensConfig.ads.toFixed(2);
            document.getElementById('scopeVal').textContent = sensConfig.scope.toFixed(2);
            
            const edpi = sensConfig.dpi * sensConfig.hip;
            const adsEdpi = edpi * sensConfig.ads;
            const cm360 = (360 / (edpi * 0.022 * 2.54)) * 2.54;
            
            document.getElementById('edpiVal').textContent = Math.round(edpi);
            document.getElementById('adsEdpiVal').textContent = Math.round(adsEdpi);
            document.getElementById('cm360Val').textContent = cm360.toFixed(1);
        }

        // Auth
        function switchAuthTab(tab) {
            currentAuthTab = tab;
            document.getElementById('tabLogin').classList.toggle('active', tab === 'login');
            document.getElementById('tabSignup').classList.toggle('active', tab === 'signup');
            
            const nameGroup = document.getElementById('nameGroup');
            const authBtn = document.getElementById('authBtn');
            
            if (tab === 'signup') {
                nameGroup.style.display = 'block';
                authBtn.textContent = 'ENLIST';
            } else {
                nameGroup.style.display = 'none';
                authBtn.textContent = 'AUTHENTICATE';
            }
            document.getElementById('errorMsg').textContent = '';
        }

        function showError(msg) {
            document.getElementById('errorMsg').textContent = msg;
        }

        function handleAuthSubmit(e) {
            e.preventDefault();
            const email = document.getElementById('email').value.trim();
            const password = document.getElementById('password').value;
            const username = document.getElementById('username').value.trim();
            
            if (!email) { showError('Enter comms ID'); return; }
            if (!password) { showError('Enter access code'); return; }
            if (currentAuthTab === 'signup' && !username) { showError('Enter callsign'); return; }
            if (password.length < 6) { showError('Code minimum 6 chars'); return; }
            
            const userId = 'op_' + btoa(email).replace(/[^a-zA-Z0-9]/g, '');
            currentUser = {
                id: userId,
                email: email,
                username: currentAuthTab === 'signup' ? username : (email.split('@')[0] || 'Operator')
            };
            
            try {
                const saved = localStorage.getItem('sniper_stats_' + currentUser.id);
                if (saved) userStats = JSON.parse(saved);
            } catch (e) {}
            
            updateStats();
            enterApp();
        }

        function loginAsGuest() {
            currentUser = {
                id: 'guest_' + Date.now(),
                email: 'field@op.local',
                username: 'Field Op'
            };
            userStats = { totalScore: 0, totalHits: 0, totalMisses: 0, gamesPlayed: 0 };
            updateStats();
            enterApp();
        }

        function enterApp() {
            document.getElementById('authScreen').classList.add('hidden');
            document.getElementById('appContainer').classList.add('active');
        }

        function logout() {
            if (currentUser && currentUser.id) {
                try {
                    localStorage.setItem('sniper_stats_' + currentUser.id, JSON.stringify(userStats));
                } catch (e) {}
            }
            currentUser = null;
            userStats = { totalScore: 0, totalHits: 0, totalMisses: 0, gamesPlayed: 0 };
            document.getElementById('authScreen').classList.remove('hidden');
            document.getElementById('appContainer').classList.remove('active');
            document.getElementById('authForm').reset();
            showError('');
        }

        function updateStats() {
            document.getElementById('totalScore').textContent = userStats.totalScore.toLocaleString();
            const total = userStats.totalHits + userStats.totalMisses;
            const acc = total > 0 ? Math.round((userStats.totalHits / total) * 100) : 0;
            document.getElementById('accuracy').textContent = acc + '%';
            document.getElementById('gamesPlayed').textContent = userStats.gamesPlayed;
        }

        // Game
        function startGame(type) {
            currentGame = type;
            document.getElementById('gameScreen').classList.add('active');
            
            const titles = {
                classic: 'ZEROING', tracking: 'TRACKING', flick: 'QUICKSCOPE', spider: 'HOSTILES'
            };
            const descs = {
                classic: 'Engage static targets at range',
                tracking: 'Maintain lead on moving target',
                flick: 'Acquire and engage before cover',
                spider: 'Multiple threats, prioritize targets'
            };
            
            document.getElementById('overlayTitle').textContent = titles[type];
            document.getElementById('overlayDesc').textContent = descs[type];
            
            resizeCanvas();
            resetGame();
        }

        function resetGame() {
            gameState = {
                score: 0, hits: 0, misses: 0, time: 60,
                startTime: null, targets: [], particles: [],
                mouseX: 0, mouseY: 0, isMouseDown: false, isScoped: false
            };
            updateGameUI();
            document.getElementById('gameOverlay').classList.remove('hidden');
            gameRunning = false;
            if (animationId) cancelAnimationFrame(animationId);
        }

        function toggleGame() {
            if (gameRunning) endGame();
            else startGameLoop();
        }

        function startGameLoop() {
            gameRunning = true;
            gameState.startTime = Date.now();
            document.getElementById('gameOverlay').classList.add('hidden');
            
            if (currentGame === 'classic') spawnClassicTargets();
            else if (currentGame === 'tracking') spawnTrackingTarget();
            else if (currentGame === 'flick') spawnFlickTarget();
            else if (currentGame === 'spider') spawnSpiderTargets();
            
            gameLoop();
            
            const timerInterval = setInterval(() => {
                if (!gameRunning) { clearInterval(timerInterval); return; }
                gameState.time--;
                updateGameUI();
                if (gameState.time <= 0) { clearInterval(timerInterval); endGame(); }
            }, 1000);
        }

        function spawnClassicTargets() {
            while (gameState.targets.length < 3) {
                gameState.targets.push({
                    x: Math.random() * (canvas.width - 100) + 50,
                    y: Math.random() * (canvas.height - 100) + 50,
                    radius: 28,
                    spawnTime: Date.now()
                });
            }
        }

        function spawnTrackingTarget() {
            gameState.targets = [{
                x: canvas.width / 2, y: canvas.height / 2,
                radius: 38, vx: (Math.random() - 0.5) * 3, vy: (Math.random() - 0.5) * 3,
                health: 100, spawnTime: Date.now()
            }];
        }

        function spawnFlickTarget() {
            if (gameState.targets.length < 2) {
                gameState.targets.push({
                    x: Math.random() * (canvas.width - 100) + 50,
                    y: Math.random() * (canvas.height - 100) + 50,
                    radius: 24, spawnTime: Date.now(), life: 1400
                });
            }
        }

        function spawnSpiderTargets() {
            while (gameState.targets.length < 4) {
                gameState.targets.push({
                    x: Math.random() * (canvas.width - 100) + 50,
                    y: Math.random() * (canvas.height - 100) + 50,
                    radius: 18, vx: (Math.random() - 0.5) * 5, vy: (Math.random() - 0.5) * 5,
                    spawnTime: Date.now()
                });
            }
        }

        function gameLoop() {
            if (!gameRunning) return;
            
            ctx.fillStyle = '#0c0e0b';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Draw tactical grid on canvas
            ctx.strokeStyle = 'rgba(74, 93, 58, 0.06)';
            ctx.lineWidth = 1;
            const gridSize = 50;
            for (let x = 0; x < canvas.width; x += gridSize) {
                ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, canvas.height); ctx.stroke();
            }
            for (let y = 0; y < canvas.height; y += gridSize) {
                ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(canvas.width, y); ctx.stroke();
            }
            
            updateTargets();
            drawTargets();
            updateParticles();
            drawParticles();
            drawTacticalCrosshair();
            
            if (currentGame === 'classic' && gameState.targets.length < 3) spawnClassicTargets();
            if (currentGame === 'flick') {
                gameState.targets = gameState.targets.filter(t => Date.now() - t.spawnTime < t.life);
                if (gameState.targets.length < 2) spawnFlickTarget();
            }
            if (currentGame === 'spider' && gameState.targets.length < 4) spawnSpiderTargets();
            
            animationId = requestAnimationFrame(gameLoop);
        }

        function updateTargets() {
            gameState.targets.forEach(target => {
                if (currentGame === 'tracking' || currentGame === 'spider') {
                    target.x += target.vx;
                    target.y += target.vy;
                    if (target.x < target.radius || target.x > canvas.width - target.radius) target.vx *= -1;
                    if (target.y < target.radius || target.y > canvas.height - target.radius) target.vy *= -1;
                    
                    if (currentGame === 'tracking') {
                        const dx = gameState.mouseX - target.x;
                        const dy = gameState.mouseY - target.y;
                        const dist = Math.sqrt(dx * dx + dy * dy);
                        if (dist < target.radius && gameState.isMouseDown) {
                            target.health -= 0.5;
                            gameState.score += 1;
                            if (target.health <= 0) {
                                createExplosion(target.x, target.y, '#8b7d5a');
                                gameState.hits++;
                                gameState.score += 50;
                                spawnTrackingTarget();
                            }
                        }
                    }
                }
            });
        }

        function drawTargets() {
            gameState.targets.forEach(target => {
                const age = Date.now() - target.spawnTime;
                
                if (currentGame === 'flick') {
                    const progress = age / target.life;
                    ctx.globalAlpha = 1 - progress;
                }
                
                // Outer ring (olive)
                ctx.beginPath();
                ctx.arc(target.x, target.y, target.radius, 0, Math.PI * 2);
                ctx.strokeStyle = currentGame === 'tracking' ? 
                    `hsl(${target.health * 1.2}, 30%, 40%)` : '#4a5d3a';
                ctx.lineWidth = 2;
                ctx.stroke();
                
                // Inner fill
                ctx.beginPath();
                ctx.arc(target.x, target.y, target.radius * 0.6, 0, Math.PI * 2);
                ctx.fillStyle = currentGame === 'tracking' ? 
                    `hsla(${target.health * 1.2}, 30%, 40%, 0.2)` : 'rgba(74, 93, 58, 0.15)';
                ctx.fill();
                
                // Center dot (tan)
                ctx.beginPath();
                ctx.arc(target.x, target.y, 3, 0, Math.PI * 2);
                ctx.fillStyle = '#8b7d5a';
                ctx.fill();
                
                ctx.globalAlpha = 1;
            });
        }

        function createExplosion(x, y, color) {
            for (let i = 0; i < 10; i++) {
                const angle = (Math.PI * 2 / 10) * i;
                const speed = Math.random() * 3 + 1;
                gameState.particles.push({
                    x: x, y: y,
                    vx: Math.cos(angle) * speed,
                    vy: Math.sin(angle) * speed,
                    life: 25, color: color,
                    size: Math.random() * 3 + 1
                });
            }
        }

        function updateParticles() {
            gameState.particles = gameState.particles.filter(p => {
                p.x += p.vx; p.y += p.vy; p.life--;
                return p.life > 0;
            });
        }

        function drawParticles() {
            gameState.particles.forEach(p => {
                ctx.beginPath();
                ctx.arc(p.x, p.y, p.size * (p.life / 25), 0, Math.PI * 2);
                ctx.fillStyle = p.color;
                ctx.globalAlpha = p.life / 25;
                ctx.fill();
            });
            ctx.globalAlpha = 1;
        }

        function drawTacticalCrosshair() {
            const x = gameState.mouseX;
            const y = gameState.mouseY;
            const size = gameState.isScoped ? 6 : 12;
            
            ctx.strokeStyle = 'rgba(184, 181, 168, 0.6)';
            ctx.lineWidth = 1;
            
            // Cross
            ctx.beginPath();
            ctx.moveTo(x - size * 2, y);
            ctx.lineTo(x - size * 0.5, y);
            ctx.moveTo(x + size * 0.5, y);
            ctx.lineTo(x + size * 2, y);
            ctx.moveTo(x, y - size * 2);
            ctx.lineTo(x, y - size * 0.5);
            ctx.moveTo(x, y + size * 0.5);
            ctx.lineTo(x, y + size * 2);
            ctx.stroke();
            
            // Center box
            ctx.strokeRect(x - size * 0.5, y - size * 0.5, size, size);
            
            // Mil dots if scoped
            if (gameState.isScoped) {
                for (let i = 1; i <= 3; i++) {
                    ctx.fillStyle = 'rgba(184, 181, 168, 0.4)';
                    ctx.fillRect(x - 1, y - size * 0.5 - i * 8, 2, 2);
                    ctx.fillRect(x - 1, y + size * 0.5 + i * 8, 2, 2);
                    ctx.fillRect(x - size * 0.5 - i * 8, y - 1, 2, 2);
                    ctx.fillRect(x + size * 0.5 + i * 8, y - 1, 2, 2);
                }
            }
        }

        function handleClick(e) {
            if (!gameRunning) return;
            
            const rect = canvas.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            
            let hit = false;
            
            if (currentGame === 'classic' || currentGame === 'flick' || currentGame === 'spider') {
                for (let i = gameState.targets.length - 1; i >= 0; i--) {
                    const target = gameState.targets[i];
                    const dx = x - target.x;
                    const dy = y - target.y;
                    const dist = Math.sqrt(dx * dx + dy * dy);
                    
                    if (dist < target.radius) {
                        hit = true;
                        gameState.hits++;
                        const timeBonus = Math.max(0, 100 - (Date.now() - target.spawnTime) / 10);
                        gameState.score += Math.floor(100 + timeBonus);
                        
                        createExplosion(target.x, target.y, '#8b7d5a');
                        gameState.targets.splice(i, 1);
                        
                        if (currentGame === 'classic') spawnClassicTargets();
                        if (currentGame === 'spider') spawnSpiderTargets();
                        break;
                    }
                }
            }
            
            if (!hit) {
                gameState.misses++;
                gameState.score = Math.max(0, gameState.score - 10);
            }
            
            updateGameUI();
        }

        function updateGameUI() {
            document.getElementById('score').textContent = gameState.score;
            document.getElementById('time').textContent = gameState.time;
            document.getElementById('hits').textContent = gameState.hits;
            document.getElementById('misses').textContent = gameState.misses;
        }

        function endGame() {
            gameRunning = false;
            if (animationId) cancelAnimationFrame(animationId);
            
            userStats.totalScore += gameState.score;
            userStats.totalHits += gameState.hits;
            userStats.totalMisses += gameState.misses;
            userStats.gamesPlayed++;
            updateStats();
            
            const total = gameState.hits + gameState.misses;
            const accuracy = total > 0 ? Math.round((gameState.hits / total) * 100) : 0;
            const avgTime = gameState.hits > 0 ? Math.round((60 - gameState.time) * 1000 / gameState.hits) : 0;
            
            document.getElementById('resultScore').textContent = gameState.score;
            document.getElementById('resultAccuracy').textContent = accuracy + '%';
            document.getElementById('resultHits').textContent = gameState.hits;
            document.getElementById('resultAvg').textContent = avgTime + 'ms';
            
            document.getElementById('resultsModal').classList.add('active');
        }

        function closeResults() {
            document.getElementById('resultsModal').classList.remove('active');
            backToMenu();
        }

        function restartGame() {
            document.getElementById('resultsModal').classList.remove('active');
            resetGame();
            startGameLoop();
        }

        function backToMenu() {
            gameRunning = false;
            if (animationId) cancelAnimationFrame(animationId);
            document.getElementById('gameScreen').classList.remove('active');
            document.getElementById('resultsModal').classList.remove('active');
        }

        // Right click to ADS (scope)
        canvas.addEventListener('contextmenu', e => {
            e.preventDefault();
            gameState.isScoped = !gameState.isScoped;
        });
        
        canvas.addEventListener('click', handleClick);
        canvas.addEventListener('mousemove', (e) => {
            const rect = canvas.getBoundingClientRect();
            gameState.mouseX = e.clientX - rect.left;
            gameState.mouseY = e.clientY - rect.top;
        });
        canvas.addEventListener('mousedown', (e) => {
            if (e.button === 0) gameState.isMouseDown = true;
        });
        canvas.addEventListener('mouseup', () => gameState.isMouseDown = false);

        window.addEventListener('beforeunload', () => {
            if (currentUser && currentUser.id) {
                try {
                    localStorage.setItem('sniper_stats_' + currentUser.id, JSON.stringify(userStats));
                } catch (e) {}
            }
        });

        // Init
        updateSens();
